// vm.c
static struct kmap {
 void *virt; //虛擬地址啟始位址
 uint phys_start; //物理地址啟始位址
 uint phys_end; //物理地址結束地址
 int perm; // 使用權限
 } kmap[] = {
 { (void*)KERNBASE, 0, EXTMEM, PTE_W}, // I/O space
 { (void*)KERNLINK, V2P(KERNLINK), V2P(data), 0}, // kern text+rodata
 { (void*)data, V2P(data), PHYSTOP, PTE_W},  // kern data+memory
 { (void*)DEVSPACE, DEVSPACE, 0, PTE_W},  // more devices
 };

代碼解釋：




















//vm.c
pde_t* setupkvm(void)
{
  pde_t *pgdir; //page directory
  struct kmap *k; 

  if((pgdir = (pde_t*)kalloc()) == 0)
    return 0;
  memset(pgdir, 0, PGSIZE);
  if (P2V(PHYSTOP) > (void*)DEVSPACE)
      panic("PHYSTOP too high");
  for(k = kmap; k < &kmap[NELEM(kmap)]; k++)
    if(mappages(pgdir, k->virt, k->phys_end - k->phys_start, 
                (uint)k->phys_start, k->perm) < 0){
      freevm(pgdir);
      return 0;
      }
  return pgdir;
}

一開始呼叫kalloc() 分配一頁給 pgdir 並把這一頁的值初始為零，接下來用 mappages() 設定各層kmap的值，這樣所有內存都有設定值了；若 mappages() 返回-1則取消virtual address配置。






















//vm.c
static int mappages(pde_t *pgdir, void *va, uint size, uint pa, int perm)
{
  char *a, *last; //地址的啟始端與結束端
  pte_t *pte; // page table entry

  a = (char*)PGROUNDDOWN((uint)va);
  last = (char*)PGROUNDDOWN(((uint)va) + size - 1);
  for(;;){
    if((pte = walkpgdir(pgdir, a, 1)) == 0)
      return -1;
    if(*pte & PTE_P)
      panic("remap");
    *pte = pa | perm | PTE_P;
    if(a == last)
      break;
    a += PGSIZE;
    pa += PGSIZE;
  }
  return 0;
}

一開始把 a 跟 last 分別對齊其頁的初始地址；進入迴圈後呼叫 walkpgdir() 獲得 size 範圍內每一頁第一個pte的位址，接著利用PTE_P判斷該pte是否已被使用，然後設定該pte映射的物理地址、權限跟flag，之後跳下一頁，直到映射完 size 大小的address。


















//vm.c
static pte_t * walkpgdir(pde_t *pgdir, const void *va, int alloc)
{
  pde_t *pde; //page directory entry
  pte_t *pgtab; //page table

  pde = &pgdir[PDX(va)];
  if(*pde & PTE_P){
    pgtab = (pte_t*)p2v(PTE_ADDR(*pde));
  } else {
    if(!alloc || (pgtab = (pte_t*)kalloc()) == 0)
      return 0;
    memset(pgtab, 0, PGSIZE);
    *pde = v2p(pgtab) | PTE_P | PTE_W | PTE_U;
  }
  return &pgtab[PTX(va)];
}

首先判斷該PTE是否存在，有的話返回該PTE；若沒有則呼叫 kalloc() 建立新的一頁並初始為零，設定flag後返回該PTE。
