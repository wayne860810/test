Chapter 4: Locking

四資工三 B10415025 洪昱翔

運作原理

由於xv6運行在多處理器上，一個function可能會在同一時間被多個CPU所執行，為了防止它們互相干擾而建立"Lock"這種機制，Lock提供了互斥功能，保證某個時間點只有一個CPU能持有，且只有持有該Lock才能使用其對應的data，以此來保護整個資料結構。

以xv6文件中的範例解釋




















struct list {
    int data;
    struct list *next;
};

struct list *list = 0;
//struct lock listlock;

void
insert(int data)
{
    //acquire(&listlock);
    struct list *l;
    l = malloc(sizeof *l);
    l->data = data;
    l->next = list;
    list = l;
    //release(&listlock);
}

假設有兩個CPU同時執行到insert這個function，執行完第16行要到第17行時，跑稍慢的CPU會將前者的資料給覆蓋掉，導致資料遺失，稱作Race Conditions。但若添加第7,12,18行，透過Lock的概念，讓同一時間只有一個CPU能夠執行這個function，其他CPU必須等資料被釋放完之後且獲取Lock才能繼續執行。
下面程式碼有更詳盡地解說。

程式流程

資料結構與功能解釋

資料結構











struct spinlock {
  uint locked;
  // 判斷lock與否，值為0則表示unlock，其餘則lock。

  // For debugging:
  char *name;        // 該lock的名稱
  struct cpu *cpu;   // 紀錄目前使用的CPU
  uint pcs[10];      // The call stack (an array of program counters)
                     // that locked the lock.
};

initlock








void
initlock(struct spinlock *lk, char *name)
{
  lk->name = name;   //給予名稱
  lk->locked = 0;    //初始化預設為unlock
  lk->cpu = 0;       //一開始沒有CPU使用
}

acquire


















void
acquire(struct spinlock *lk)
{
  pushcli(); // 避免中斷導致deadlock
  if(holding(lk)) // 判斷此lock是否被同一個CPU取得
    panic("acquire"); //給予error提醒

  // 判斷該lock是否被其他CPU所佔據，1則表示有需要再等待，0則表示沒有可獲取lock
  while(xchg(&lk->locked, 1) != 0);

  // 在取得lock後，以確保資料不遺失
  __sync_synchronize();

  lk->cpu = mycpu(); //紀錄目前使用的CPU
  getcallerpcs(&lk, lk->pcs);
}

release



















void
release(struct spinlock *lk)
{
  if(!holding(lk)) // 若lock沒有被占用
    panic("release"); //給予error提醒

  // 設定為初始值
  lk->pcs[0] = 0;
  lk->cpu = 0;

  // 在釋放lock後，以確保資料不遺失
  __sync_synchronize();

  // 將lk-locked設為0
  asm volatile("movl $0, %0" : "+m" (lk->locked) : );

  popcli();
}

getcallerpcs


















// Record the current call stack in pcs[] by following the %ebp chain.
void
getcallerpcs(void *v, uint pcs[])
{
  uint *ebp;
  int i;

  ebp = (uint*)v - 2;
  for(i = 0; i < 10; i++){
    if(ebp == 0 || ebp < (uint*)KERNBASE || ebp == (uint*)0xffffffff)
      break;
    pcs[i] = ebp[1];     // saved %eip
    ebp = (uint*)ebp[0]; // saved %ebp
  }
  for(; i < 10; i++)
    pcs[i] = 0;
}

holding







int
holding(struct spinlock *lock)
{
  // 判斷目前獲取lock的CPU是否和原本相同
  return lock->locked && lock->cpu == mycpu();
}

pushcli












void
pushcli(void)
{
  int eflags;

  eflags = readeflags(); // 取得目前即時狀態
  cli();
  if(mycpu()->ncli == 0) // 若為0則存入目前狀態
    mycpu()->intena = eflags & FL_IF;
  mycpu()->ncli += 1; // 層數+1
}

popcli











void
popcli(void)
{
  if(readeflags()&FL_IF)
    panic("popcli - interruptible");
  if(--mycpu()->ncli < 0)
    panic("popcli");
  if(mycpu()->ncli == 0 && mycpu()->intena)
    sti();
}
