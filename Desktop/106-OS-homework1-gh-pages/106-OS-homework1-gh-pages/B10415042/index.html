第四章 Lock

為了說明Lock的必要性，以下面的程式為範例。

struct list{
    int data;
    struct list *next;
};

struct list *list = 0;

void
insert(int data)
{
    struct list *l;
    l = malloc(sizeof *l);
    l->data = data;
    l->next = list;
    list = l;
}

該程式看起來雖然沒有什麼錯誤，但在多執行緒的情況下，若是複數的執行緒都正好執行到，

    l->next = list;

或者，

    list = l;

的話，會因為race hazard造成指標的錯亂，為此，我們需要一個鎖(Lock)，以確保同一時間下只有一個insert在執行。

因此，我們有了1501的這個struct。

1501 struct spinlock {
1502 uint locked; // 是否鎖著
1503
1504 // For debugging:
1505 char *name; // Name of lock.
1506 struct cpu *cpu; // 哪個cpu拿著鎖
1507 uint pcs[10]; // 存在stack指標的陣列
1508 // that locked the lock.
1509 };

為了保證上鎖的程式不會發生開頭提到的race hazard，我們必須保證上鎖程序是不可分割的(atomic)，xv6以下面的xchg函式達成。

0569 xchg(volatile uint *addr, uint newval)
0570 {
0571 uint result;
0572
0573 // The + in "+m" denotes a read−modify−write operand.
0574 asm volatile("lock; xchgl %0, %1" :
0575 "+m" (*addr), "=a" (result) :
0576 "1" (newval) :
0577 "cc");
0578 return result;
0579 }

取得鎖的程式則是利用xchg來達成，若是回傳1則代表鎖著繼續loop，若是回傳0則可以取得，並在取得後紀錄資訊，方便之後debug等動作。

1574 acquire(struct spinlock *lk)
1575 {
1576 pushcli(); // 關掉中斷以防止鎖死
1577 if(holding(lk))
1578 panic("acquire");
1579
1580 // xchg的操作不可分割
1581 while(xchg(&lk−>locked, 1) != 0)
1582 ;
1583
1584 // 插入memory barrier保證讀寫
1585 //
1586 //
1587 __sync_synchronize();
1588
1589 // 紀錄資訊方便debug
1590 lk−>cpu = mycpu();
1591 getcallerpcs(&lk, lk−>pcs);
1592 }

lock的使用

而使用鎖的時機，則是如同最前文的第一個例子，若是當一個cpu正在處理一個變數時，有另一個cpu可能會讀取或是寫入那個變數的話，那就有必要使用鎖。
又或是一個invariant涉及多記憶體位址，那它亦需要鎖來保證他的不變。

但能不使用鎖則進可能不使用，因為這會拖累執行效率。

若要保護kernel，亦可使用一個鎖，在進入kernel時上鎖，在離開kernel時解鎖，這可以應用於單核程式執行在多核處理器的時候，一般稱之Giant lock。這個作法失去的平行的效能，更好的作法是用更好的鎖讓kernel能讓多個cpu使用

xv6本身只使用了幾個簡單的鎖。

lock的順序

現在假設程式foo1以及foo2如下

foo1()
{
……
……
acquire(&lock_one);
acquire(&lock_two);
……
……
release(&lock_one);
release(&lock_two);
}

foo2()
{
……
……
acquire(&lock_two);
acquire(&lock_one);
……
……
release(&lock_two);
release(&lock_one);
}

雖然看似沒有問題，但若是foo1執行完

acquire(&lock_one);

的時候，foo2執行完

acquire(&lock_two);

的話，則會陷入鎖死(Deadlock)的情況，foo1無法取得lock_two，造成無法釋放lock_one；foo2無法取得lock_one，造成無法釋放lock_two，兩邊都陷入死輪迴。

為了避免鎖死(Deadlock)的情況，上鎖的順序很是重要，必須所有需要一樣鎖的程式碼都以相同的順序去call。

xv6中的例子來說。就是ideintr

4302 // Interrupt handler.

4304 ideintr(void)
4305 {
4306 struct buf *b;
4307
4308 // First queued buffer is the active request.
4309 acquire(&idelock);
4310
4311 if((b = idequeue) == 0){
4312 release(&idelock);
4313 return;
4314 }
4315 idequeue = b−>qnext;
4316
4317 // Read data if needed.
4318 if(!(b−>flags & B_DIRTY) && idewait(1) >= 0)
4319 insl(0x1f0, b−>data, BSIZE/4);
4320
4321 // Wake process waiting for this buf.
4322 b−>flags |= B_VALID;
4323 b−>flags &= ~B_DIRTY;

4325
4326 // Start disk on next buf in queue.
4327 if(idequeue != 0)
4328 idestart(idequeue);
4329
4330 release(&idelock);
4331 }

可以看到程式碼中原本就含有idelock，而其中的再呼叫的wakeup()則

2962 // Wake up all processes sleeping on chan.

2965 {
2966 acquire(&ptable.lock);
2967 wakeup1(chan);
2968 release(&ptable.lock);
2969 }

也需要ptable.lock，xv6中還有很多用到sleep以及wakeup的函式都是如此。

中斷

xv6中鎖還能用來防止系統中斷程式執行和另一個非中斷的cpu執行程式的race hazard，以下面的中斷程式的片段為例子

3413 switch(tf−>trapno){
3414 case T_IRQ0 + IRQ_TIMER:
3415 if(cpuid() == 0){
3416 acquire(&tickslock);
3417 ticks++;
3418 wakeup(&ticks);
3419 release(&tickslock);
3420 }

若是同時另一個執行緒在執行下面的程式

則其中的ticks則會重複使用發生錯誤。

看上面的程式碼，ideintr也會需要idelock，但idelock已經被中斷前的iderw拿去，iderw則又被ideintr中斷無法釋放idelock，程式陷入鎖死狀態。

整理下，就是
1.執行中斷程式時可能變數會和其他執行緒的程式衝突
2.中斷時若保有中斷程式需要的lock會造成鎖死

為此，中斷時，程序絕不能擁有中斷處理程式需要的lock，xv6在這裡更進一步的要求，中斷時，不能擁有任何lock。 xv6使用以下兩函式來達成。

1654 void
1655 pushcli(void)
1656 {
1657 int eflags;
1658
1659 eflags = readeflags();
1660 cli();
1661 if(mycpu()−>ncli == 0)
1662 mycpu()−>intena = eflags & FL_IF;
1663 mycpu()−>ncli += 1;
1664 }
1665

1666 void
1667 popcli(void)
1668 {
1669 if(readeflags()&FL_IF)
1670 panic("popcli − interruptible");

1672 panic("popcli");

1674 sti();
1675 }

1582 ;
1583
1584 // 插入memory barrier保證讀寫
1585 //
1586 //
1587 __sync_synchronize();
1588
1589 // 紀錄資訊方便debug
1590 lk−>cpu = mycpu();

1592 }

並在執行release的時候執行popcli(1622)解除屏蔽，這邊同上理由，為防止鎖死最後才執行。

1600 // Release the lock.
1601 void
1602 release(struct spinlock *lk)
1603 {
1604 if(!holding(lk))
1605 panic("release");
1606
1607 lk−>pcs[0] = 0;
1608 lk−>cpu = 0;
1609
1610 // Tell the C compiler and the processor to not move loads or stores
1611 // past this point, to ensure that all the stores in the critical
1612 // section are visible to other cores before the lock is released.
1613 // Both the C compiler and the hardware may re−order loads and
1614 // stores; __sync_synchronize() tells them both not to.
1615 __sync_synchronize();
1616
1617 // Release the lock, equivalent to lk−>locked = 0.
1618 // This code can’t use a C assignment, since it might
1619 // not be atomic. A real OS would use C atomics here.
1620 asm volatile("movl $0, %0" : "+m" (lk−>locked) : );
1621
1622 popcli();
1623 }

Memory ordering

由於處理器經常使用亂序執行，為此我們需要memory barrier，這裡應用到的是__sync_synchronize()

1574 acquire(struct spinlock *lk)
1575 {
1576 pushcli(); // 關掉中斷以防止鎖死
1577 if(holding(lk))
1578 panic("acquire");
1579
1580 // xchg的操作不可分割
1581 while(xchg(&lk−>locked, 1) != 0)
1582 ;
1583
1584 // 插入memory barrier保證讀寫
1585 //
1586 //
1587 __sync_synchronize();
1588
1589 // 紀錄資訊方便debug
1590 lk−>cpu = mycpu();
1591 getcallerpcs(&lk, lk−>pcs);
1592 }

__sync_synchronize()會插入memory barrier，保證前面的讀寫完畢才進行後面的動作。
這樣就能保證不會出錯。

總結

現在lock也在不斷的進化，目標是能讓上鎖的時候，同步的效率也盡可能不低落。
如果要使用lock，必須確定好哪些東西是需要上鎖的，因為容易遺漏所以需要多次確認，最好是使用些工具來確認。
實現非atomic的鎖雖然可能，但代價太過高昂。然而如此代價依舊不低，因此許多os使用的資料結構和演算法均與鎖無關，並在這些算法中盡量避免atomic的操作。
