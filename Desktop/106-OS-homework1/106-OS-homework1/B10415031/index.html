1.運作原理

一個Page的Page DirectoryTable (PDT) 包含 1024 個 Page Table Page (PTP) Page Directory Entry。
每個 PDP 包含1024個32bit的PTE。

Page hardware將virtual address前10個bit 用於指定PDE，如果該PDE的PTE_P置位則繼續，否則fault。
Page hardware將virtual address中間10個bit 用於指定PTE，如果該PTE的PTE_P置位則繼續，否則fault。

每個PTE中含有標示来表示對應的地址的訪問權限
PTE_P : 表示當前PTE是否有效，如果無效的PTE會導致fault
PTE_W : 可寫，不可寫的話只能讀和執行
PTE_U : 用戶程序可以訪問，否則只有kernel process能訪問
每個process都有一個獨立的page，process切換時候需要切換page.

Pde_t* setupkvm(void)
{
	Pde_t *pgdir;
	Struct kmap *k;
	if((pgdir = (pde_t*)kalloc())==0)
		return 0;
	memset(pgdir,0,PGSIZE);
	for(k=kmap;k<&kmap[NELEM(kmap)];k++)
		if(mappages(pgdir,k->virt,kphys_end-k->phys_start,
                                    (unit)k->phys_start,k->perm<0)
	return0;
    return pgdir;
}//setupkvm一開始用kalloc分配一些page來存Page Directory
 //setupkvm掃描kmap的每個元素，並且呼叫mappages

mappages





















static int mappages(pde_t *pgdir, void *va, uint size, uint pa, int perm)
{
  char *a, *last;
  pte_t *pte;

  a = (char*)PGROUNDDOWN((uint)va);
  last = (char*)PGROUNDDOWN(((uint)va) + size - 1);
  for(;;){
    if((pte = walkpgdir(pgdir, a, 1)) == 0)
      return -1;
    if(*pte & PTE_P)
      panic("remap");//如果該位被置位則說明返回的PTE已對應page table。這在page建立時是不會出現的，會panic。
    *pte = pa | perm | PTE_P;
    if(a == last)
      break;
    a += PGSIZE;
    pa += PGSIZE;
  }
  return 0; 
}//mappages()會利用PGROUNDDOW進行位址的對齊，接著對KMAP元素指向的MEMORY內所有的PAGE執行walkpgdir()程式

walkpgdir



















static pte_t * walkpgdir(pde_t *pgdir, const void *va, int alloc)
{
  pde_t *pde;
  pte_t *pgtab;

  pde = &pgdir[PDX(va)];
  if(*pde & PTE_P){
    pgtab = (pte_t*)p2v(PTE_ADDR(*pde));
  } else {
    if(!alloc || (pgtab = (pte_t*)kalloc()) == 0)
      return 0; //確認所有PTE_P 的bits是0
    
    memset(pgtab, 0, PGSIZE);
   
    *pde = v2p(pgtab) | PTE_P | PTE_W | PTE_U;
  }
  return &pgtab[PTX(va)];
} //walkpgdir()會去查詢Page Directory中與va相關的PDE索引。如果PDE已經被放置了，說明對應的PT已存在，否則透過kalloc()分配一個PT

void kinit(void) 
{ 
  extern int end; 
  uint mem; 
  char *start; 
  initlock(&kalloc_lock, "kalloc"); 
  start = (char*) &end; 
  start = (char*) (((uint)start + PAGE) & ~(PAGE−1)); 
  mem = 256; // 假設電腦有 256 pages的RAM 
  cprintf("mem = %d\n", mem * PAGE); 
  kfree(start, mem * PAGE); 
}//使用這個函數是因為main中的code大多不能使用lock，也不能存4mb以上的資料.

kfree




































kfree(char *v, int len) 
{ 
struct run *r, *rend, **rp, *p, *pend; 
   if(len <= 0 || len % PAGE) 
    panic("kfree"); 

   // Fill with junk to catch dangling refs. 
   memset(v, 1, len); //首先將被釋放的內存的每一個字節設為1
  acquire(&kalloc_lock); 
  p = (struct run*)v; 
  pend = (struct run*)(v + len); 
     rend = (struct run*)((char*)r + r−>len); 
   if(r <= p && p < rend) 
     panic("freeing free page"); 
    if(pend == r){  // replace r with p 
       p−>len = len + r−>len; 
       p−>next = r−>next; 
       *rp = p; 
      goto out;
     } 
     if(rend == p){  // r next to p: replace p with r 
      r−>len += len; 
       if(r−>next && r−>next == pend){  // r now next to r−>next? 
         r−>len += r−>next−>len; 
         r−>next = r−>next−>next;
     } 
      goto out; 
   } 
   // 在r之前insert p
  p−>len = len; 
  p−>next = r; 
  *rp = p; 
  out: 
  release(&kalloc_lock); //kalloc 移除並回傳第一個element到free list。
}

sys_sbrk(void)
{
  int addr;
  int n;

  if(argint(0, &n) < 0)
    return -1;
  addr = cp->sz;
  if(growproc(n) < 0)
    return -1;
  return addr;
}//sbrk是一個system call用來增加或減少memory
 //這個system call implemented by the function growproc

growproc


















//如果n是positive growproc，分配一個以上的physical pages然後maps他們在process address的最上面
//如果n是negative，growproc unmap一個以上的pages然後free掉對應的physical 
growproc(int n)page
{
  char *newmem;

  newmem = kalloc(cp->sz + n);
  if(newmem == 0)
    return -1;
  memmove(newmem, cp->mem, cp->sz);
  memset(newmem + cp->sz, 0, n);
  kfree(cp->mem, cp->sz);
  cp->mem = newmem;
  cp->sz += n;
  usegment();
  return 0;
}

(d) exec

exec














































































































//exec 是創建地址空間中用戶的部分的system call
exec(char *path, char **argv)
{
  char *s, *last;
  int i, off;
  uint argc, sz, sp, ustack[3+MAXARG+1];
  struct elfhdr elf;
  struct inode *ip;
  struct proghdr ph;
  pde_t *pgdir, *oldpgdir;

  if((ip = namei(path)) == 0)//透過namei打開二進制的文件
    return -1;
  ilock(ip);
  pgdir = 0;

  // Check ELF header
  // exec的第一步是檢查文件是否包含ELF二進制代碼。
  //一個ELF二進制文件是以4個“magic number”開頭，即 0x7F，“E”，“L”，“F”，或者 ELF_MAGIC。
  //如果ELF header中包含正確的magic number，exec 就會認為該二進制文件的结構是正確的。
  if(readi(ip, (char*)&elf, 0, sizeof(elf)) < sizeof(elf))
    goto bad;//透過readi把內容讀出 
  if(elf.magic != ELF_MAGIC)
    goto bad;

  if((pgdir = setupkvm(kalloc)) == 0)
    goto bad;////透過allocuvm為每個ELF分配memory

  // Load program into memory.

  // leave first page inaccessible (to make NULL ref fail)
  sz = PGSIZE-1;

  for(i=0, off=elf.phoff; i<elf.phnum; i++, off+=sizeof(ph)){
    if(readi(ip, (char*)&ph, off, sizeof(ph)) != sizeof(ph))
      goto bad;
    if(ph.type != ELF_PROG_LOAD)
      continue;
    if(ph.memsz < ph.filesz)
      goto bad;
    if((sz = allocuvm(pgdir, sz, ph.vaddr + ph.memsz)) == 0)
      goto bad;//allocates a new page一個新的沒有user mapping的page table 
    if(loaduvm(pgdir, (char*)ph.vaddr, ip, ph.off, ph.filesz) < 0)
    // loads each segment into memory
    //loaduvm透過walkpgdir來找到寫入ELF segment的memory的physical address
          goto bad;
  }
  iunlockput(ip);
  ip = 0;

  // Allocate two pages at the next page boundary.
  // Make the first inaccessible.  Use the second as the user stack.
  //allocuvm會檢查請求分配的virtual address是否在KERNBASE之下
  sz = PGROUNDUP(sz);
  if((sz = allocuvm(pgdir, sz, sz + 2*PGSIZE)) == 0)
    goto bad;
  clearpteu(pgdir, (char*)(sz - 2*PGSIZE));
  sp = sz;

  // Push argument strings, prepare rest of stack in ustack.
  for(argc = 0; argv[argc]; argc++) {
    if(argc >= MAXARG)
      goto bad;
    sp = (sp - (strlen(argv[argc]) + 1)) & ~3;
    if(copyout(pgdir, sp, argv[argc], strlen(argv[argc]) + 1) < 0)
      goto bad;
    ustack[3+argc] = sp;
  }
  ustack[3+argc] = 0;

  ustack[0] = 0xffffffff;  // fake return PC
  ustack[1] = argc;
  ustack[2] = sp - (argc+1)*4;  // argv pointer

  sp -= (3+argc+1) * 4;
  if(copyout(pgdir, sp, ustack, (3+argc+1)*4) < 0)
    goto bad;

  // Save program name for debugging.
  for(last=s=path; *s; s++)
    if(*s == '/')
      last = s+1;
  safestrcpy(proc->name, last, sizeof(proc->name));

  // hook up the old shared memory to the new addr space
  // (no need to change refcount, because the old ref goes away with
  // this exec, so it's +1 then -1 => 0 change
  if (proc->shared) {
    extern void mappages(pde_t *pgdir, void *va, uint size, uint pa, int perm);
    mappages(pgdir, (char *)SHARED_V, PGSIZE, v2p(proc->shared->page), PTE_W|PTE_U);
  }

  // Commit to the user image.
  oldpgdir = proc->pgdir;
  proc->pgdir = pgdir;
  proc->sz = sz;
  proc->tf->eip = elf.entry;  // main
  proc->tf->esp = sp;
  switchuvm(proc);
  freevm(oldpgdir);
  return 0;

 bad:
  if(pgdir)
    freevm(pgdir);
  if(ip)
    iunlockput(ip);
  return -1;
}
