acquire




















acquire(struct spinlock *lk) // 用來鎖定lock的function
{
    pushcli(); // 上鎖期間禁止interrupts直到解鎖
    if(holding(lk)) // 確認lock沒有被這一個CPU重複上鎖
        panic("acquire"); // 遭此CPU重複上鎖時，輸出錯誤訊息，供除錯使用

    // 利用不會被中斷的function xchg進行上鎖動作
    while(xchg(&lk−>locked, 1) != 0) // 重複執行直到確認已被此CPU上鎖
    ;

    // Tell the C compiler and the processor to not move loads or stores
    // past this point, to ensure that the critical section’s memory
    // references happen after the lock is acquired.
    __sync_synchronize();

    // 紀錄上鎖的CPU和pcs資訊讓錯誤發生時可以找到線索
    lk−>cpu = mycpu(); // 紀錄上鎖者是此CPU
    getcallerpcs(&lk, lk−>pcs); // 紀錄pcs資訊
}

// 因此要在同一行內上鎖和確認尚未被上鎖，故使用xchg function












xchg(volatile uint *addr, uint newval) // 用來進行上鎖動作的function
{
    uint result; //用來記錄原本在記憶體內的資料，因為此function是利用兩塊記憶體交換來達到assign的功能

    // The + in "+m" denotes a read−modify−write operand.
    asm volatile("lock; xchgl %0, %1" :
    "+m" (*addr), "=a" (result) :
    "1" (newval) :
    "cc"); // 利用組合語言將兩塊記憶體內容交換，即可在assign的同時確認原本在內部之值
    return result;
}

release























release(struct spinlock *lk) //用來解除lock鎖定的function
{
     if(!holding(lk)) //確認lock是不是被本CPU上鎖
         panic("release"); //若已經解鎖或是非本CPU上鎖，輸出錯誤資訊，供除錯使用
    
     lk−>pcs[0] = 0; // 清除pcs[0]資訊
     lk−>cpu = 0; // 清除上鎖者資訊

     // Tell the C compiler and the processor to not move loads or stores
     // past this point, to ensure that all the stores in the critical
     // section are visible to other cores before the lock is released.
     // Both the C compiler and the hardware may re−order loads and
     // stores; __sync_synchronize() tells them both not to.
     __sync_synchronize();

     // Release the lock, equivalent to lk−>locked = 0.
     // This code can’t use a C assignment, since it might
    // not be atomic. A real OS would use C atomics here.
    asm volatile("movl $0, %0" : "+m" (lk−>locked) : ); // 利用組合語言將lock解鎖，但真正的OS這裡會使用C的atomics function

    popcli(); // 重新允許上鎖時禁止的interrupts
}

資料結構:









struct spinlock {
    uint locked; // 紀錄是否已經被鎖定，若為0則尚未鎖定、非0即已被鎖定
    // 其他資訊被用來除錯使用
    char *name; // 紀錄此lock的名稱
    struct cpu *cpu; // 紀錄目前被哪個cpu鎖定
    uint pcs[10]; // The call stack (an array of program counters)
    // that locked the lock.
};
